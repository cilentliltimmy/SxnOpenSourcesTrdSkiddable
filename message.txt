local function getClosestEnemy()
	local char = LocalPlayer.Character
	if not char then return nil end
	local hrp = char:FindFirstChild("HumanoidRootPart")
	if not hrp then return nil end

	local closest, dist = nil, math.huge
	for _, plr in pairs(Players:GetPlayers()) do
		if plr ~= LocalPlayer and (not plr.Team or plr.Team ~= LocalPlayer.Team) then
			local c = plr.Character
			local h = c and c:FindFirstChild("HumanoidRootPart")
			if h then
				local d = (hrp.Position - h.Position).Magnitude
				if d < dist then
					dist = d
					closest = plr
				end
			end
		end
	end
	return closest
end

local function addHighlight(player)
	if Highlights[player] then return end
	local char = player.Character
	if not char then return end
	local highlight = Instance.new("Highlight")
	highlight.Adornee = char
	highlight.FillColor = Color3.fromRGB(255,0,0)
	highlight.OutlineColor = Color3.fromRGB(255,255,255)
	highlight.Parent = game.CoreGui
	Highlights[player] = highlight
end

local function removeHighlight(player)
	if Highlights[player] then
		Highlights[player]:Destroy()
		Highlights[player] = nil
	end
end

local lastShot = 0
local cooldown = 0

local Toggle = Tab:CreateToggle({
    Name = "Auto-Kill The Closest Person",
    CurrentValue = false,
    Flag = "AutoKillToggle",
    Callback = function(state)
        if ToggleConn then
            ToggleConn:Disconnect()
            ToggleConn = nil
        end

        for p,_ in pairs(Highlights) do
            removeHighlight(p)
        end

        if state then
            ToggleConn = RunService.RenderStepped:Connect(function()
                local target = getClosestEnemy()
                if not target or not target.Character then return end

                local hrp = target.Character:FindFirstChild("HumanoidRootPart")
                local char = LocalPlayer.Character
                local gun = char and char:FindFirstChild("PaintballGun")
                local handle = gun and (gun:FindFirstChild("Handle") or gun:FindFirstChildWhichIsA("BasePart"))

                if hrp and gun and handle then
                    addHighlight(target)
                    Camera.CFrame = CFrame.lookAt(Camera.CFrame.Position, hrp.Position)

                    if tick() - lastShot >= cooldown then
                        lastShot = tick()
                        local shootCFrame = CFrame.lookAt(handle.Position, hrp.Position)
                        gun.Fire:FireServer(shootCFrame)
                    end
                end
            end)
        end
    end
})

        -- reference to in-game player name folder
        local season = RS:FindFirstChild("Season")
        if not season then return end

        local playersFolder = season:FindFirstChild("Players")
        if not playersFolder then return end

        local function getInGameName(player)
            local data = playersFolder:FindFirstChild(player.Name)
            if data and data.Value ~= "" then
                return data.Value
            end
            return player.Name
        end

        for i, p1 in ipairs(Players:GetPlayers()) do
            for j, p2 in ipairs(Players:GetPlayers()) do
                if j > i then
                    local ok, areFriends = pcall(function()
                        return p1:IsFriendsWith(p2.UserId)
                    end)

                    if ok and areFriends then
                        local in1 = getInGameName(p1)
                        local in2 = getInGameName(p2)

                        print(in1.." ("..p1.Name..") is friends with "..in2.." ("..p2.Name..")")
                    end
                end
            end
        end
   end,
})
